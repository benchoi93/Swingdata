\section{Supplementary Figures}
\label{sec:appendix}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig1_radar_behavioral_profile.pdf}
\caption{Multi-dimensional behavioral profiles by operating mode (TUB, STD, ECO). Values are normalized to [0,1] range for each outcome. TUB mode dominates on harsh events (aggressive riding) while ECO dominates on cruise fraction (stable riding). Speed CV is nearly identical across modes.}
\label{fig:radar}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig2_forest_gee_coefficients.pdf}
\caption{GEE coefficient forest plot for all primary outcomes (200,000-trip subsample). TUB is the reference mode. Points show estimated coefficients; error bars show 95\% confidence intervals. Safety outcomes (harsh events) show the strongest mode differentiation.}
\label{fig:forest}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=0.9\textwidth]{v2/fig3_shap_safety_harsh_event.pdf}
\caption{SHAP summary plot for the harsh event classification model (LightGBM, AUC = 0.734). TUB mode indicator dominates with mean $|\text{SHAP}| = 0.49$, confirming that mode selection is the primary determinant of harsh event occurrence.}
\label{fig:shap}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig4_did_multi_outcome.pdf}
\caption{Multi-outcome DiD coefficients with Bonferroni correction (KEY FIGURE). Red bars indicate Bonferroni-significant effects; orange bars indicate nominal significance ($p < 0.05$); gray bars indicate non-significant effects. Only harsh acceleration and deceleration survive Bonferroni correction.}
\label{fig:did_coef}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig5_demand_response.pdf}
\caption{Demand response to TUB ban by city treatment intensity. Trip counts and active users indexed to February 2023 baseline, stratified by TUB share quartiles. The December decline is seasonal (all quartiles drop similarly), not treatment-driven.}
\label{fig:demand}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig6_compensation_cohens_d.pdf}
\caption{Compensation test: Cohen's $d$ effect sizes for mode-switcher within-user DiD across all behavioral margins. All effects are negligible ($|d| < 0.2$), indicating zero behavioral compensation on any unconstrained margin.}
\label{fig:compensation}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig7_experience_trajectories.pdf}
\caption{Experience-mediated behavioral escalation. Left panel: TUB adoption rate by trip number. Right panels: outcome trajectories for all modes (dark line) vs.\ STD/ECO only (blue dashed). The gap between lines quantifies TUB-mediated effects: 60--72\% for safety outcomes, 18--27\% for dynamics.}
\label{fig:experience}
\end{figure}

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{v2/fig8_survival_multi_endpoint.pdf}
\caption{Multi-endpoint Kaplan-Meier survival curves stratified by TUB usage. Left: first harsh event. Center: first high-CV trip (note: TUB is protective due to mechanical speed capping reducing CV). Right: first speeding event (non-TUB users median never reached).}
\label{fig:survival}
\end{figure}
